  <script>

    // ---------- Helpers ----------

    const $ = (s) => document.querySelector(s);

    const chips = $('#chips');

    const historyWrap = $('#historyWrap');

    const historyList = $('#historyList');

    const startNum = $('#startNum');

    const endNum = $('#endNum');

    const pickCount = $('#pickCount');

    const drawBtn = $('#drawBtn');

    const copyBtn = $('#copyBtn');

    const clearBtn = $('#clearBtn');

    const reshuffleBtn = $('#reshuffleBtn');

    const HISTORY_MAX = 8;

    const swalBase = {

      buttonsStyling: false,

      customClass: {

        popup: 'rounded-3 shadow-lg !bg-white !text-slate-800',

        confirmButton: 'btn btn-primary',

        cancelButton: 'btn btn-outline-secondary ms-2',

        denyButton: 'btn btn-warning ms-2',

        htmlContainer: 'mt-2',

      }

    };

    function secureRandInt(maxExclusive){

      if (window.crypto && crypto.getRandomValues) {

        const arr = new Uint32Array(1);

        const limit = Math.floor(0x100000000 / maxExclusive) * maxExclusive;

        let r; do { crypto.getRandomValues(arr); r = arr[0]; } while (r >= limit);

        return r % maxExclusive;

      }

      // Fallback

      return Math.floor(Math.random() * maxExclusive);

    }

    function shuffle(a){

      for (let i=a.length-1;i>0;i--){

        const j = secureRandInt(i+1);

        [a[i], a[j]] = [a[j], a[i]];

      }

      return a;

    }

    function validateInputs(){

      const s = parseInt(startNum.value, 10);

      const e = parseInt(endNum.value, 10);

      const c = parseInt(pickCount.value, 10);

      if ([s,e,c].some(Number.isNaN)) return {ok:false, msg:'숫자를 올바르게 입력해 주세요.'};

      if (s > e) return {ok:false, msg:'시작 번호가 끝 번호보다 클 수 없습니다.'};

      const total = e - s + 1;

      if (c < 1 || c > total) return {ok:false, msg:`뽑을 인원은 1 이상, ${total} 이하로 설정해 주세요.`};

      return {ok:true, s, e, c, total};

    }

    function draw(){

      const v = validateInputs();

      if (!v.ok){

        Swal.fire({...swalBase, icon:'warning', title:'입력값 확인', text:v.msg});

        return;

      }

      Swal.fire({

        ...swalBase,

        title: '추첨 중…',

        html: '<div class="d-flex justify-content-center py-3"><div class="spinner-border text-primary" role="status"></div></div>',

        allowOutsideClick: false, showConfirmButton:false, didOpen: () => Swal.showLoading(), timer: 900

      }).then(() => {

        const pool = Array.from({length: v.total}, (_, i) => v.s + i);

        const picks = shuffle(pool).slice(0, v.c).sort((a,b)=>a-b);

        renderChips(picks);

        pushHistory(picks);

        const html = `<div class="flex flex-wrap gap-2 items-center justify-center">${picks.map(n =>

          `<span class=\"px-3 py-1 rounded-xl bg-emerald-200 text-emerald-900 fw-bold\">${n}</span>`

        ).join('')}</div>`;

        Swal.fire({

          ...swalBase,

          icon: 'success',

          title: '오늘의 당번',

          html,

          confirmButtonText: '좋아요'

        });

      });

    }

    function renderChips(nums){

      chips.innerHTML = '';

      nums.forEach((n, i) => {

        const d = document.createElement('div');

        d.className = 'chip';

        d.style.animationDelay = (i*60)+'ms';

        d.textContent = n;

        chips.appendChild(d);

      });

    }

    function pushHistory(picks){

      const li = document.createElement('li');

      li.textContent = `${new Date().toLocaleTimeString([], {hour:'2-digit', minute:'2-digit'})} — ${picks.join(', ')}`;

      historyList.prepend(li);

      while (historyList.children.length > HISTORY_MAX) historyList.lastChild.remove();

      historyWrap.hidden = false;

    }

    function copyResult(){

      const vals = Array.from(chips.children).map(c => c.textContent);

      if (!vals.length){

        Swal.fire({...swalBase, icon:'info', title:'복사할 결과가 없습니다', text:'먼저 추첨을 진행해 주세요.'});

        return;

      }

      const text = vals.join(', ');

      const doToast = (icon, title) => Swal.fire({

        ...swalBase,

        toast:true, position:'top-end', timer:1800, showConfirmButton:false, icon, title

      });

      if (navigator.clipboard){

        navigator.clipboard.writeText(text).then(() => doToast('success', '결과가 복사되었습니다')).catch(() => doToast('error', '복사에 실패했습니다'));

      } else {

        // Fallback

        const ok = window.prompt('아래 텍스트를 복사하세요', text);

        if (ok !== null) doToast('success', '복사 완료');

      }

    }

    function clearAll(){

      Swal.fire({

        ...swalBase,

        icon:'question', title:'초기화할까요?',

        text:'결과와 기록이 모두 삭제됩니다.',

        showCancelButton:true, confirmButtonText:'초기화', cancelButtonText:'취소'

      }).then(res => {

        if (res.isConfirmed){

          chips.innerHTML = '';

          historyList.innerHTML = '';

          historyWrap.hidden = true;

          Swal.fire({...swalBase, toast:true, position:'top', timer:1200, showConfirmButton:false, icon:'success', title:'초기화 완료'});

        }

      });

    }

    function clampCount(){

      const v = validateInputs();

      if (!v.ok) return; // silently ignore until valid

      if (parseInt(pickCount.value,10) > v.total) pickCount.value = v.total;

    }

    // ---------- Events ----------

    drawBtn.addEventListener('click', draw);

    reshuffleBtn.addEventListener('click', draw);

    copyBtn.addEventListener('click', copyResult);

    clearBtn.addEventListener('click', clearAll);

    startNum.addEventListener('input', clampCount);

    endNum.addEventListener('input', clampCount);
